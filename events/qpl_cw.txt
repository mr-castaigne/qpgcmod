		name = qpl_cw.19.a
		drop_cosmetic_tag = yes
		promote_character = {
			character = CSC_riseup_collective
			ideology = cypherpunk_socialism
		}
		start_civil_war = {
			ruling_party = communism
			ideology = fascism
		}
		random_other_country = {
			limit = {
				original_tag = CSC
				has_government = fascism
			}
			set_cosmetic_tag = CSC_futurist
		}
	}
	option = { #this is fine
		name = qpl_cw.19.b
		if = {
			limit = {
				has_global_flag = qfs_appeared
			}
			QFS = {
				annex_country = {
					target = CSC
				}
			}
		}
		else = {
			hidden_effect = {
				QPL = { change_tag_from = CSC }
			}
			QPL = {
				annex_country = {
					target = CSC
				}
			}
		}
	}
}
